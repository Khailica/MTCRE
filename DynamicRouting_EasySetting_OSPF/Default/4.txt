set [ find default-name=ether1 ] disable-running-check=no name=ether1-anticlockwise
set [ find default-name=ether2 ] disable-running-check=no name=ether2-diagonal
set [ find default-name=ether3 ] disable-running-check=no name=ether3-clockwise
set [ find default-name=ether4 ] disable-running-check=no name=ether4-LAN
set [ find default-name=ether5 ] disable-running-check=no
set [ find default-name=ether6 ] disable-running-check=no
set [ find default-name=ether7 ] disable-running-check=no
set [ find default-name=ether8 ] disable-running-check=no
set [ find default-name=ether9 ] disable-running-check=no name=ether9-INTERNET
set [ find default-name=ether10 ] disable-running-check=no name=ether10-management
/interface wireless security-profiles
set [ find default=yes ] supplicant-identity=MikroTik
/ip address
add address=3.3.3.1/24 interface=ether1-anticlockwise network=3.3.3.0
add address=4.4.4.2/24 interface=ether3-clockwise network=4.4.4.0
/ip firewall mangle
add action=mark-routing chain=prerouting in-interface=ether3-clockwise new-routing-mark=backup-route passthrough=yes protocol=icmp src-address=!1.1.1.1
/ip firewall nat
add action=src-nat chain=srcnat out-interface=ether1-anticlockwise src-address=40.40.40.0/24 to-addresses=3.3.3.1
add action=src-nat chain=srcnat out-interface=ether2-diagonal src-address=40.40.40.0/24 to-addresses=5.5.5.4
add action=src-nat chain=srcnat out-interface=ether3-clockwise src-address=40.40.40.0/24 to-addresses=4.4.4.2
/ip route
add distance=1 dst-address=1.1.1.0/24 gateway=3.3.3.2 routing-mark=backup-route
add distance=1 dst-address=2.2.2.0/24 gateway=3.3.3.2 routing-mark=backup-route
add distance=1 dst-address=1.1.1.0/24 gateway=4.4.4.1
add distance=2 dst-address=1.1.1.0/24 gateway=3.3.3.2
add distance=1 dst-address=2.2.2.0/24 gateway=4.4.4.1
add distance=2 dst-address=2.2.2.0/24 gateway=3.3.3.2
/system identity
set name=R4
